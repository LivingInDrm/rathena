aldeba_in,96,181,4	script	卡夫拉服务	113,{
	cutin "kafra_05",2;
	mes "[卡夫拉·莉莉亚]";
	mes "嗯...? 欢迎来到卡夫拉公司总部。您需要什么吗？";
	next;
	switch(select("保存:使用储存:租用推车:取消")) {
	case 1:
		mes "[卡夫拉·莉莉亚]";
		mes "您的重生点已在卡夫拉公司总部保存。谢谢。";
		next;
		savepoint "aldeba_in",96,179,1,1;
		mes "[卡夫拉·莉莉亚]";
		mes "请利用遍布米德加尔的卡夫拉服务。感谢您光临卡夫拉总部。";
		close2;
		break;
	case 2:
		if (Zeny < 20) {
			mes "[卡夫拉·莉莉亚]";
			mes "抱歉，您似乎没有20个仙币来支付储存访问费...";
			close3;
		}
		mes "[卡夫拉·莉莉亚]";
		mes "虽然这个设施专门用于卡夫拉员工的培训和行政功能，";
		mes "但我会为您访问储存。";
		next;
		set Zeny, Zeny-20;
		set RESRVPTS, RESRVPTS + (20/5);
		mes "[卡夫拉·莱拉]";
		mes "如果您希望使用任何卡夫拉服务，";
		mes "请在未来向值班的卡夫拉员工询问。感谢您的光临。";
		callfunc("F_CheckKafCode");    //check your storage password, if set
		close2;
		openstorage;
		break;
	case 3:
		mes "[卡夫拉·莱拉]";
		mes "抱歉，我现在不在值班。如果可以的话我会帮助您，";
		mes "但实际上我没有可用的推车。";
		mes "您可以向其他卡夫拉员工寻求帮助。";
		next;
		mes "[卡夫拉·莱拉]";
		mes "卡夫拉员工遍布米德加德大陆，";
		mes "您应该能在阿尔德巴兰的外面找到很多。";
		close2;
		break;
	case 4:
		mes "[卡夫拉·莱拉]";
		mes "^666666*呼...*^000000 太好了，因为我现在正好在休息。";
		mes "选择“取消”是个明智的决定。";
		close2;
		break;
	}
	cutin "",255;
	end;
}

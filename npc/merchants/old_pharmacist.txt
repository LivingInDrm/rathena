alberta_in,16,28,4	script	药剂师	61,{
	if (checkweight(1201,1) == 0) {
		mes "- 等一下！ -";
		mes "- 你现在携带的物品 -";
		mes "- 太多了。 -";
		mes "- 请稍后回来 -";
		mes "- 在你把一些物品放入卡普拉仓库之后。 -";
		close;
	}
	mes "[老人药剂师]";
	mes "嗯嗯...";
	mes "你来这里有什么事...？";
	next;
	switch(select("制作药水:对话.:混合信息:取消")) {
	case 1:
		if (MaxWeight - Weight < 5000) {
			mes "[老人药剂师]";
			mes "你怎么能带这么多东西！";
			mes "别贪心，只带你需要的东西！";
			next;
			mes "[老人药剂师]";
			mes "你太重了，无法从我这里领取药水...";
			mes "先去把一些物品存入你的仓库！";
			close;
		} else {
			mes "[老人药剂师]";
			mes "你准备好所有的东西了吗？你想要哪一个？";
			next;
			switch(select("红色药水.:橙色药水.:黄色药水.:白色药水.:蓝色药水.:绿色药水.:其实，我什么都不想要。")) {
			case 1:
				callsub L_Making,507,3,501;
			case 2:
				mes "[老人药剂师]";
				if (countitem(507) < countitem(713)) {
					set .@max,countitem(507);
				}
				else if (countitem(508) < countitem(713)) {
					set .@max,countitem(508);
				}
				else {
					set .@max,countitem(713);
				}
				if ((countitem(507) < 1) || (countitem(508) < 1) || (countitem(713) == 0)) {
					mes "你这个小鬼！你期待什么？！空手来这里。啧！";
					mes "滚开！";
					close;
				}
				if (Zeny < 3) {
					mes "你这个小鬼！你期待什么？！空手来这里。啧！";
					mes "滚开！";
					close;
				}
				mes "多少？";
				next;
				switch(select("尽可能多地制作.:我想选择一个数量.:其实，我什么都不想要。")) {
				case 1:
					if ((countitem(507) < .@max) || (countitem(508) < .@max) || (countitem(713) < .@max) || (Zeny < .@max*3)) {
						mes "[老人药剂师]";
						mes "你这个小鬼！你连所有材料都没有，还想让我给你做药水？！";
						close;
					}
					set Zeny,Zeny-(.@max*5);
					delitem 507,.@max; //Red_Herb
					delitem 508,.@max; //Yellow_Herb
					delitem 713,.@max; //Empty_Bottle
					getitem 502,.@max; //Orange_Potion
					break;
				case 2:
					mes "[老人药剂师]";
					mes "那么请在100以下选择一个数字。如果你不想要任何东西，";
					mes "就输入'0'。根据你拥有的材料，你大约可以制作";
					next;
					input .@amount,0,101;
					if (.@amount == 0) {
						mes "[老人药剂师]";
						mes "你能不能下定决心？！";
						close;
					}
					if (.@amount > 100) {
						mes "[老人药剂师]";
						mes "你耳朵聋了吗？我说少于100！";
						close;
					}
					if ((countitem(507) < .@amount) || (countitem(508) < .@amount) || (countitem(713) < .@amount) || (Zeny < .@amount*3)) {
						mes "[老人药剂师]";
						mes "你这个小鬼！你连所有材料都没有，还想让我给你做药水？！";
						close;
					}
					next;
					set Zeny,Zeny-(.@amount*5);
					delitem 507,.@amount; //Red_Herb
					delitem 508,.@amount; //Yellow_Herb
					delitem 713,.@amount; //Empty_Bottle
					getitem 502,.@amount; //Orange_Potion
					break;
				case 3:
					mes "[老人药剂师]";
					mes "什么？！";
					mes "咕噜...";
					mes "呸！";
					mes "滚开！";
					close;
				}
				mes "[老人药剂师]";
				mes "给你。都做好了，你可以拿走。但记住！滥用药物可不好。";
				close;
			case 3:
				callsub L_Making,508,10,503;
			case 4:
				callsub L_Making,509,20,504;
			case 5:
				callsub L_Making,510,30,505;
			case 6:
				callsub L_Making,511,3,506;
			case 7:
				mes "[老人药剂师]";
				mes "什么？！";
				mes "咕噜...";
				mes "呸！";
				mes "滚开！";
				close;
			}
		}
	case 2:
		mes "[老人药剂师]";
		mes "通过药物，你可以提高一个人的再生能力。但是，它们只有在一定程度上有效。*叹气* 我开始想起年轻时的日子了。我一定是老了。";
		next;
		mes "[老人药剂师]";
		mes "无论如何，药水就是药水。没有更多也没有更少。";
		close;
	case 3:
		mes "[老人药剂师]";
		mes "嗯...";
		mes "你们年轻人真让人烦。但，既然你问了，我就解释一下。";
		next;
		mes "[老人药剂师]";
		mes "草药单独使用效果很好，但如果你用我的特殊技巧和技能把它们做成药水，效果会更好。";
		next;
		mes "[老人药剂师]";
		mes "如果你热切而礼貌地请求，我会为你制作。但，不是免费的... 不过别担心，我只收取少量费用，所以并不贵。";
		next;
		mes "[老人药剂师]";
		mes "红色药水 - ^0098E52 红色草药，1个空瓶，2金币费用。^000000";
		mes "橙色药水 - ^0098E51 红色草药，1个黄色草药，1个空瓶，5金币费用。^000000";
		mes "黄色药水 - ^0098E52 黄色草药，1个空瓶，10金币。^000000";
		next;
		mes "[老人药剂师]";
		mes "白色药水 - ^0098E52 白色草药，1个空瓶，20金币费用。^000000";
		mes "蓝色药水 - ^0098E52 蓝色草药，1个空瓶，30金币费用。^000000";
		mes "绿色药水 - ^0098E52 绿色草药，1个空瓶，3金币费用。^000000";
		close;
	case 4:
		mes "[老人药剂师]";
		mes "真是个无聊的人。如果你有话要说，为什么不说出来？！";
		close;
	}
	
L_Making:
	.@item_req = getarg(0);
	.@req_amount = getarg(1);

	mes "[老人药剂师]";
	if ((countitem(.@item_req)/2) < countitem(713)) {
		.@max = countitem(.@item_req) /2;
	} else {
		.@max = countitem(713);
	}
	if (countitem(.@item_req) < 2 || countitem(713) == 0) {
		mes "你这个小鬼！你期待什么？！空手来这里。啧！";
		mes "滚开！";
		close;
	}
	if (Zeny < 3) {
		mes "你这个小鬼！你期待什么？！空手来这里。啧！";
		mes "滚开！";
		close;
	}
	mes "多少？";
	next;
	switch(select("尽可能多地制作.:我想选择一个数量.:其实，我什么都不想要。")) {
	case 1:
		if ((countitem(.@item_req) < .@max*2) || countitem(713) < .@max || (Zeny < .@max*.@req_amount)) {
			mes "[老人药剂师]";
			mes "你这个小鬼！你连所有材料都没有，还想让我给你做药水？！";
			close;
		}
		Zeny = Zeny - (.@max*.@req_amount);
		delitem .@item_req,.@max*2;
		delitem 713,.@max; //Empty_Bottle
		getitem getarg(2),.@max; 
		break;
	case 2:
		mes "[老人药剂师]";
		mes "那么请在100以下选择一个数字。如果你不想要任何东西，";
		mes "就输入'0'。根据你拥有的材料，你大约可以制作";
		next;
		input .@amount;
		if (.@amount == 0) {
			mes "[老人药剂师]";
			mes "你能不能下定决心？！";
			close;
		}
		if (.@amount > 100) {
			mes "[老人药剂师]";
			mes "你耳朵聋了吗？我说少于100！";
			close;
		}
		if ((countitem(.@item_req) < .@amount*2) || countitem(713) < .@amount || (Zeny < .@amount*.@req_amount)) {
			mes "[老人药剂师]";
			mes "你这个小鬼！你连所有材料都没有，还想让我给你做药水？！";
			close;
		}
		Zeny = Zeny - (.@max*.@req_amount);
		delitem .@item_req,.@amount*2;
		delitem 713,.@amount; //Empty_Bottle
		getitem getarg(2),.@amount; 
		break;
	case 3:
		mes "[老人药剂师]";
		mes "什么？！";
		mes "咕噜...";
		mes "呸！";
		mes "滚开！";
		close;
	}
	mes "[老人药剂师]";
	mes "给你。这些都做好了，方便你带走。但记住！滥用药物是不好的。";
	close;
}

		getitem getarg(3), rand(1,getarg(4));
		if (rand(10)) {
			next;
			mes "You picked up "+getarg(2)+" Mora Coins next to the remains.";
			specialeffect2 EF_STEAL;
			getitem 6380,getarg(2); //Mora_Coin
		}
		close;
	}
	mes "You see traces of recent digging.";
	close;
}
